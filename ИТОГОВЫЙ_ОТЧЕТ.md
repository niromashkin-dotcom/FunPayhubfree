#### 2.2 Реализация `CacheManager`

- Файл `runtime/cache/cache_manager.py` был пустым стабом.
- Реализован полноценный `CacheManager` с TTL, записью в JSON-файлы и фоновым обновлением.

```python
# runtime/cache/cache_manager.py (фрагмент)
class CacheManager:
    def __init__(self, cache_dir: str = "data/cache", default_ttl: int = 300):
        self._cache: dict[str, tuple[Any, float]] = {}
        self._cache_dir = Path(cache_dir)
        self._cache_dir.mkdir(parents=True, exist_ok=True)
        self._default_ttl = default_ttl

    def get(self, key: str) -> Any | None:
        value, ts = self._cache.get(key, (None, 0))
        if value is not None and time.time() - ts < self._default_ttl:
            return value
        return None

    def set(self, key: str, value: Any, ttl: int | None = None) -> None:
        self._cache[key] = (value, time.time())
        self._persist(key, value)
```

# web/seller_api.py (фрагмент)
def _toggle_lots(svc, supplier, deactivate: bool):
    lots_data = svc.get_my_lots(force_refresh=True)
    markers = ["[AS#", "[GB#", "[HB#", "[KS#"]
    toggled = 0
    for lot in lots_data.get("lots", []):
        title = lot.get("title", "")
        if not any(m in title for m in markers):
            continue
        lot_id = lot.get("id")
        if lot_id:
            svc.toggle_lot_active(lot_id, not deactivate, dry_run=False)
            toggled += 1
    return toggled

#### 2.8 Исправление `SellerService.create_lot`

**Ключевое исправление для реального создания лотов на FunPay**

- Было: `LotFields` создавался без `subcategory` и `node_id`, из-за чего FunPay отвечал `{"done":false,"error":"Категория не найдена."}`.
- Стало: метод ищет подкатегорию по `subcategory_id` (если передан) или по `category_id`, выставляет `node_id` и передает объект `subcategory` в `LotFields`.
- Также добавлены английские поля формы, которые FunPay требует для всех новых лотов.

```python
# runtime/seller_service.py (фрагмент create_lot)
def create_lot(self, lot_data: dict) -> dict:
    with self._lock:
        acc = self._get_account()
        if acc is None:
            return {"ok": False, "error": self._last_error or "Нет авторизации"}
        try:
            title = str(lot_data.get("title", ""))
            descr = str(lot_data.get("description", ""))
            price = float(lot_data.get("price") or 0)
            amount = int(lot_data.get("amount") or 1)
            if not title or not price:
                return {"ok": False, "error": "Название и цена обязательны"}

            from FunPayAPI import types as _fp_types
            fields = {
                "fields[summary][ru]": title,
                "fields[summary][en]": title,          # ← добавлено
                "fields[desc][ru]": descr,
                "fields[desc][en]": descr,              # ← добавлено
                "price": str(price),
                "amount": str(amount),
                "active": "on",
            }

            subcat = None
            node_id = 0
            # Приоритет: subcategory_id → category_id
            subcat_id = int(lot_data.get("subcategory_id") or 0)
            if subcat_id:
                for cat in acc.get_sorted_categories().values():
                    subcat = cat.get_subcategory(_fp_types.SubCategoryTypes.COMMON, subcat_id)
                    if subcat:
                        break
            if subcat is None and lot_data.get("category_id"):
                category = acc.get_category(int(lot_data["category_id"]))
                if category:
                    subs = [s for s in category.get_subcategories()
                            if getattr(s, "type", None) is _fp_types.SubCategoryTypes.COMMON]
                    if subs:
                        subcat = subs[0]

            if subcat is not None:
                node_id = subcat.id
                fields["node_id"] = str(node_id)

            lot_fields = _fp_types.LotFields(0, fields, subcategory=subcat)
            acc.save_lot(lot_fields)
            return {"ok": True}
        except Exception as e:
            return {"ok": False, "error": _safe_error(e)}
```
